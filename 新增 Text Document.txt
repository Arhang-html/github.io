<!-- 新增食譜 -->
<div style="background:white;padding:20px;border:1px solid #ccc;border-radius:10px;">
    <h2>新增食譜</h2>
    <label>名稱（英文）*</label><input type="text" id="new-name-en">
    <label>名稱（中文）</label><input type="text" id="new-name-zh">
    <label>完成份量</label><input type="text" id="new-servings" placeholder="e.g. 8份">
    <label>材料（一行一個）</label><textarea id="new-ingredients" rows="4"></textarea>
    <label>步驟（一行一個）</label><textarea id="new-steps" rows="6"></textarea>
    <label>照片連結（Google Drive 分享連結，可留空）</label><input type="text" id="new-photo" placeholder="https://drive.google.com/...">
    <button onclick="addNewRecipe()">儲存食譜</button>
</div>

<script>
const WEB_APP_URL = 'https://docs.google.com/spreadsheets/d/1ZkhKqg8tfkhMvllL0YVQ-EZn0sJwk-pFHElqvLazBIw/edit?usp=drive_link';

let recipes = [];

async function loadRecipes() {
    try {
        const res = await fetch(WEB_APP_URL);
        recipes = await res.json();
        populateBothLists();
    } catch(e) {
        alert('無法連線到 Google Sheet，請確認 Web App URL 正確且已部署為「任何人」');
    }
}

function populateBothLists() {
    const search = document.getElementById('search-input').value.toLowerCase();
    let filtered = recipes.filter(r => 
        r.name_en.toLowerCase().includes(search) || 
        r.name_zh.toLowerCase().includes(search)
    );
    filtered.sort((a,b) => a.name_en.localeCompare(b.name_en));

    const left = document.getElementById('recipe-list-left');
    const right = document.getElementById('recipe-list-right');
    left.innerHTML = '<option value="">-- 選擇左邊 --</option>';
    right.innerHTML = '<option value="">-- 選擇右邊 --</option>';

    filtered.forEach(r => {
        const name = r.name_zh || r.name_en;
        const opt = `<option value="${r.id}">${name}</option>`;
        left.innerHTML += opt;
        right.innerHTML += opt;
    });
}

async function addNewRecipe() {
    const name_en = document.getElementById('new-name-en').value.trim();
    if (!name_en) return alert('必須輸入英文名稱！');

    const payload = {
        action: 'add',
        name_en: name_en,
        name_zh: document.getElementById('new-name-zh').value.trim() || name_en,
        servings: document.getElementById('new-servings').value.trim() || '1份',
        ingredients: document.getElementById('new-ingredients').value.trim().split('\n').map(i=>i.trim()).filter(Boolean),
        steps: document.getElementById('new-steps').value.trim().split('\n').map(s=>s.trim()).filter(Boolean),
        photo: document.getElementById('new-photo').value.trim()
    };

    await fetch(WEB_APP_URL, {method:'POST', body: JSON.stringify(payload)});
    alert('已儲存到 Google Sheet！');
    loadRecipes();   // 即時刷新
    clearForm();
}

function clearForm() {
    document.getElementById('new-name-en').value = '';
    document.getElementById('new-name-zh').value = '';
    document.getElementById('new-servings').value = '';
    document.getElementById('new-ingredients').value = '';
    document.getElementById('new-steps').value = '';
    document.getElementById('new-photo').value = '';
}

// 顯示左右食譜（略，與之前一樣）
function displayLeft() { renderDisplay('left'); }
function displayRight() { renderDisplay('right'); }
function renderDisplay(side) {
    const id = parseInt(document.getElementById(`recipe-list-${side}`).value);
    const recipe = recipes.find(r => r.id === id);
    if (!recipe) return;
    const container = document.getElementById(`display-${side}`);
    let html = `<img src="${recipe.photo}" style="max-width:100%;display:block;">`;
    html += `<h2>${recipe.name_zh || recipe.name_en}</h2>`;
    html += `<h3>完成份量：</h3><p>${recipe.servings}</p>`;
    html += `<h3>材料：</h3><ul>${recipe.ingredients.map(i=>`<li>${i}</li>`).join('')}</ul>`;
    html += `<h3>步驟：</h3><ol>${recipe.steps.map(s=>`<li>${s}</li>`).join('')}</ol>`;
    html += `<button class="delete-btn" onclick="deleteRecipe(${recipe.id})">🗑 刪除</button>`;
    container.innerHTML = html;
}

async function deleteRecipe(id) {
    if (!confirm('確認刪除？')) return;
    await fetch(WEB_APP_URL, {method:'POST', body: JSON.stringify({action:'delete', id:id})});
    alert('已刪除並同步到 Sheet');
    loadRecipes();
}

function changeLanguage() { populateBothLists(); }

window.onload = loadRecipes;
</script>
